    s=1.0 /(1+np.exp(-z))
    
    ### END THE CODE ###

def cost(y_pred, y_label):
    """
    
        y_pred
        y_label
    
        cst: 
    """   
    ### START THE CODE ###
    
    cst=-np.sum((y_label*np.log(y_pred))+(1-y_label)*np.log(1-y_pred))
    
    ### END THE CODE ###
    
    return cst

    """
    
        X

        theta_0:

def accuracy(Y_hat, Y_label):
    """
    
        Y_hat 0 or 1
        Y_label
    
        acc:

def gradient_descent(x_bt, y_bt, theta, theta_0, learning_rate):
    """batch
    
        x_bt:
        y_bt:
        theta,theta_0
    
        theta,theta_0
        
        theta, theta_0
        theta_0theta
    """ 
    ### START THE CODE ###

    pred=predict(x_bt,theta,theta_0)

    temp=x_bt.shape[0]
    gradient=np.dot(pred-y_bt,x_bt)/temp
    gradient_0=np.sum(pred-y_bt)/temp

    theta-=learning_rate*gradient
    theta_0-=learning_rate*gradient_0
    ### END THE CODE ###
    
    return theta,theta_0

def train(X, Y, theta, theta_0, max_iter, batch_size, learning_rate):
    train_cost = []
    test_cost = []
    train_acc = []
    test_acc = []

    for epoch in range(max_iter):
        X, Y = shuffle(X, Y) # 
        for i in range(int(np.floor(len(Y)/batch_size))): #batch
            x_bt = X[i*batch_size:(i+1)*batch_size] #32*106
            y_bt = Y[i*batch_size:(i+1)*batch_size] #32label
            
            #batchtheta,theta_0
            theta,theta_0 = gradient_descent(x_bt,y_bt,theta,theta_0,learning_rate)
        
        #epoch
        Y_pred = predict(X, theta, theta_0)
        Y_hat = np.round(Y_pred)
        train_acc.append(accuracy(Y_hat, Y))
        train_cost.append(cost(Y_pred, Y)/num_train)
        #epoch
        Y_test_pred = predict(X_test, theta, theta_0)
        Y_test_hat = np.round(Y_test_pred)
        test_acc.append(accuracy(Y_test_hat, Y_test))
        test_cost.append(cost(Y_test_pred, Y_test)/num_test)
        
    return train_cost,test_cost,train_acc,test_acc
